def _pack
  class_name = self.class.to_s
  self.__pack = Aqua::Storage.new
  self.__pack.id = @id if @id
  self.__pack[:_rev] = _rev if _rev 
  self.__pack[:class] = class_name
  _pack_properties
  _pack_singletons
  __pack
end
