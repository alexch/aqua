def method_missing( method, *args )
  if respond_to?( :commit ) 
    raise NoMethodError
  else
    Aqua.set_storage_engine # to default, currently CouchDB   
    send( method.to_sym, eval(args.map{|value| "'#{value}'"}.join(', ')) ) # resend!
  end  
end
