def reload( mask_exceptions = true ) 
  if id.nil?
    if mask_exceptions
      false
    else
      raise ObjectNotFound, "#{self.class} instance must have an id to be reloaded"
    end    
  else
    begin
      _reload
    rescue Exception => e 
      if mask_exceptions
        false
      else 
        raise e
      end    
    end      
  end  
end
